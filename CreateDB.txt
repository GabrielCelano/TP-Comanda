CREATE TABLE Pedidos (
  ID INT AUTO_INCREMENT PRIMARY KEY,
  CódigoAlfanumérico VARCHAR(5) NOT NULL,
  IDCliente INT,
  IDMesa VARCHAR(5),
  Estado ENUM('pendiente', 'en preparación', 'listo para servir') NOT NULL,
  TiempoEstimadoFinalización TIMESTAMP,
  EmpleadoTomóPedido VARCHAR(255),
  EmpleadoPreparóPedido VARCHAR(255),
  EmpleadoSirvióPedido VARCHAR(255),
  FOREIGN KEY (IDCliente) REFERENCES Clientes(ID),
  FOREIGN KEY (IDMesa) REFERENCES Mesas(Código)
);

CREATE TABLE Encuestas (
  ID INT AUTO_INCREMENT PRIMARY KEY,
  IDMesa VARCHAR(5),
  PuntuaciónMesa INT CHECK(PuntuaciónMesa >= 1 AND PuntuaciónMesa <= 10),
  PuntuaciónRestaurante INT CHECK(PuntuaciónRestaurante >= 1 AND PuntuaciónRestaurante <= 10),
  PuntuaciónMozo INT CHECK(PuntuaciónMozo >= 1 AND PuntuaciónMozo <= 10),
  PuntuaciónCocinero INT CHECK(PuntuaciónCocinero >= 1 AND PuntuaciónCocinero <= 10),
  Comentario VARCHAR(66),
  FOREIGN KEY (IDMesa) REFERENCES Mesas(Código)
);
